function batch_rename
    set prefix $argv[1]
    set counter 1

    for file in *.$argv[2]
        if test -f "$file"
            set new_name (printf "%s_%02d.%s" $prefix $counter $argv[2])
            mv "$file" "$new_name"
            echo "é‡å‘½å: $file -> $new_name"
            set counter (math $counter + 1)
        end
    end
end
